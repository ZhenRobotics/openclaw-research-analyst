🚀 如何使用
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# 方式 1: 异步模式（推荐）
python3 scripts/cn_market_report.py --async

# 方式 2: 同步模式（回退）
python3 scripts/cn_market_report.py --sync

# 方式 3: 默认模式（基于 .env.cn_market 配置）
python3 scripts/cn_market_report.py

🔧 配置说明
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
编辑 .env.cn_market 文件：

CN_MARKET_USE_ASYNC=true        # 启用异步模式（默认）
CN_MARKET_USE_ASYNC=false       # 禁用异步，使用同步模式

更多配置选项详见 .env.cn_market 文件注释
